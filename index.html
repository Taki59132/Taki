10242 Fourth Point !! 第六周作業二

發表於 2017-10-18 | 分類於 Java作業 | | 閱讀次數: 
/*Author : 蔡宗翰
 *學號：410577003
 *Date : 2017/10/18
 *第六周作業二  10242:Fourth Point !!
 */
import java.util.*;

public class A410577003_6_2 {
    /*
     * 解法：輸入為兩個邊，故將前兩個x,y作為底邊，再看相同的座標是右邊還是左邊的來決定第四個座標的位置。
     */
    public static void getForthPoint(double[] tmp, double[][] coori, int preIndex, int aftIndex) {
        double ansX = tmp[0] + (coori[preIndex][0] - coori[aftIndex][0]);
        double ansY = tmp[1] + (coori[preIndex][1] - coori[aftIndex][1]);
        System.out.printf("%.3f %.3f", ansX, ansY);
        System.out.println();
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        while (sc.hasNextDouble()) {
            double[][] coori = new double[3][2];
            for (int j = 0; j < 2; j++) {// 輸入兩個座標當作底邊
                coori[j][0] = sc.nextDouble();
                coori[j][1] = sc.nextDouble();
            }
            double[] tmp = new double[2];// 因為不知道這個是不是與前兩個座標一樣，故為tmp
            tmp[0] = sc.nextDouble();
            tmp[1] = sc.nextDouble();
            if (tmp[0] == coori[0][0] && tmp[1] == coori[0][1]) {// tmp跟左邊的座標相同
                tmp[0] = sc.nextDouble();// 將tmp置換成第三個座標
                tmp[1] = sc.nextDouble();
                getForthPoint(tmp, coori, 1, 0);
            } else if (tmp[0] == coori[1][0] && tmp[1] == coori[1][1]) {// tmp跟右邊的座標相同
                tmp[0] = sc.nextDouble();// 將tmp置換成第三個座標
                tmp[1] = sc.nextDouble();
                getForthPoint(tmp, coori, 0, 1);

            } else {// tmp 並不跟前兩個座標相同，一樣還是要判斷是左邊的還是右邊的，才知道方向。
                double sameX = sc.nextDouble();
                double sameY = sc.nextDouble();
                if (sameX == coori[0][0] && sameY == coori[0][1]) {// same
                                                                    // 跟左邊座標相同
                    getForthPoint(tmp, coori, 1, 0);
                } else if (sameX == coori[1][0] && sameY == coori[1][1]) {// same跟右邊座標相同
                    getForthPoint(tmp, coori, 0, 1);
                }
            }
        }
        sc.close();
    }
}
